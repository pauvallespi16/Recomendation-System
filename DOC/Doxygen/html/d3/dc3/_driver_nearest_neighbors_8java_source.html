40 private static Map<String, Atributo> fillAtributos(Map<String, Atributo> atributos, int j) throws Exception {
41 INOUT.write("Introduce el nombre de un atributo del item "+j+": ");
42 String name = INOUT.readline();
43 INOUT.writeln("Introduce el tipo del atributo #" + j + " de entre los siguientes:");

45 String tipo = INOUT.readline();
46 while (!tipo.equals("numerico") && !tipo.equals("booleano") && !tipo.equals("categorico") && !tipo.equals("categorico multiple")) {
47 INOUT.writeln("Dato incorrecto, por favor introduzca uno de los tipos de datos dados.");
48 tipo = INOUT.readline();
49 }
50 switch (tipo) {
51 case "numerico":

58 INOUT.readline();
59 AtributoNumerico atributoNumerico = new AtributoNumerico(d.toString());
60 atributoNumerico.setValorNormalizado(d);
61 atributos.put(name, atributoNumerico);
62 break;

64 case "booleano":
65 INOUT.write("Introduce un booleano (true/false): ");
66 String b = INOUT.readline();
67 AtributoBoolean atributoBooleano = new AtributoBoolean(b);
68 atributos.put(name, atributoBooleano);
69 break;
70
71 case "categorico":
72 INOUT.write("Introduce un string: ");
73 String s = INOUT.readline();

75 atributos.put(name, atributoCategorico);
76 break;
77
78 case "categorico multiple":
79 INOUT.write("Introduce varias palabras sin espacios y separadas por punto y coma (;): ");
80 String sm = INOUT.readline();
81 AtributoCategoricoMultiple atributoCategoricoMultiple = new AtributoCategoricoMultiple(sm.toString());
82 atributos.put(name, atributoCategoricoMultiple);

97 for (int i=0; i<n; i++){

131 }
