261 public double computeEuclideanDistance(Map<Integer, Double> vals, Map<Integer, Double> valsP, Integer centroid) {
262 double sum = 0.0;
263 if (valsP == null){
264 for (Integer key : centroides.get(centroid).getImaginaryUsers().keySet()){
265 double c = centroides.get(centroid).getImaginaryUsers().get(key);
266 if (vals.containsKey(key)){
267 double v = vals.get(key);
268 sum += Math.pow((v - c), 2);
269 }
270 }
271 }
272
273 else {
274 for (Integer key : centroides.get(centroid).getImaginaryUsers().keySet()){
275 if (vals.containsKey(key) && valsP.containsKey(key)) {
276 double v1 = vals.get(key);
277 double v2 = valsP.get(key);
278 sum += Math.pow((v1 - v2), 2);
279 }
280 else if (vals.containsKey(key)) sum += Math.pow(vals.get(key), 2);
281 else if (valsP.containsKey(key)) sum += Math.pow(valsP.get(key), 2);
282 }
283 }
284 return Math.sqrt(sum);
285 }

524 private Double sillhouetteCoef(Usuario u, Integer cluster){
525 if (clusterMap.get(cluster).size() == 1) return 0.0;
526
527 double sim = similarity(u, cluster);
528 double dis = dissimilarity(u, cluster);
529 return (dis - sim / Math.max(dis, sim));
