124 private BufferedReader in = new BufferedReader(
125 new InputStreamReader(System.in));
126 private char[] buffer = new char[80];
127 private int typed= 0, used= 0;
128
129 private char readkey() throws Exception
130 { if (used == typed)
131 { String line = readkeyline();
132 if (line.equals(EOF))
133 { typed = 0; buffer[typed] = eof; }
134 else
135 { typed = line.length();
136 for (int i = 0; i < typed; i++)
137 buffer[i] = line.charAt(i);
138 buffer[typed] = nl;
139 }
140 typed = typed + 1; used = 0;
141 }
142 char ch = buffer[used];
143 if (ch == cr) ch = nl;
144 used = used + 1; return ch;
145 }

147 private String readkeyline() throws Exception
148 { String line = in.readLine();
149 return (line == null?"":line);
150 }
