300 private static String validRegex() {
301 String otherThanQuote = " [^\"] ";
302 String quotedString = String.format(" \" %s* \" ", otherThanQuote);
303 return String.format("(?x) " + // enable comments, ignore white spaces
304 ", " + // match a comma
305 "(?= " + // start positive look ahead
306 " (?: " + // start non-capturing group 1
307 " %s* " + // match 'otherThanQuote' zero or more times
308 " %s " + // match 'quotedString'
309 " )* " + // end group 1 and repeat it zero or more times
310 " %s* " + // match 'otherThanQuote'
311 " $ " + // match the end of the string
312 ") ", // stop positive look ahead
313 otherThanQuote, quotedString, otherThanQuote);
314 }

349 public static void updateContents(Item it) {
350 items.addItem(it);
351 itemsTable.put(it.getOriginalId(), it.getId());
352 for (int i = 0; i < headersItems.length; i++) {
353 String header = headersItems[i];
354 Atributo at = it.getAtributos().get(header);
355
356 // separamos por casos según el tipo
357 if (at instanceof AtributoNumerico) {
358 auxNormalizeNumeros[i][0] = Math.min(auxNormalizeNumeros[i][0], (Double) at.getValor());
359 auxNormalizeNumeros[i][1] = Math.max(auxNormalizeNumeros[i][1], (Double) at.getValor());
360 } else if (at instanceof AtributoFecha) {
361 auxNormalizeFechas[i][0] = Math.min(auxNormalizeFechas[i][0], (Long) at.getValor());
362 auxNormalizeFechas[i][1] = Math.max(auxNormalizeFechas[i][1], (Long) at.getValor());
363 } else if (at instanceof AtributoCategoricoMultiple) {
364 Set<String> set = ((AtributoCategoricoMultiple) at).getValor();
365 for (String s : set) { if (!bitsTable.containsKey(s)) { bitsTable.put(s, bitIndex); bitIndex++; } }
366 ((AtributoCategoricoMultiple) at).setBitSetSize(bitsTable.size());
367 for (String s : set) ((AtributoCategoricoMultiple) at).setBitSetValue(bitsTable.get(s));
368 } else if (at instanceof AtributoTexto) {
369 Map<String, Integer> freq = ((AtributoTexto) at).getValor();
370 Map<String, Integer> idfMap = IDF.get(header);
371 for (Map.Entry<String, Integer> word : freq.entrySet()) {
372 if (idfMap.containsKey(word.getKey())) {
373 int n = idfMap.get(word.getKey());
374 idfMap.put(word.getKey(), n + 1);
375 } else idfMap.put(word.getKey(), 1);
376 }
377 IDF.put(headersItems[i], idfMap);
378 }
379 }
380 normalizarItems();

389 public static void normalizarItems() {
390 ArrayList<Item> arr = new ArrayList(items.getItems().values());
391
392 for (int i = 0; i < arr.size(); i++) {
393 Item it = arr.get(i);
394 Map<String, Atributo> atts = it.getAtributos();
395
396 // iteramos sobre todos los elementos de la cabecera
397 for (int j = 0; j < headersItems.length; j++) {
398 Atributo atr = atts.get(headersItems[j]);
399
400 // si el elemento es texto
401 if (atr instanceof AtributoTexto) {
402 Map<String, Integer> idfMap = IDF.get(headersItems[j]);
403 Map<String, Integer> freq = ((AtributoTexto) atr).getValor();
404 for (Map.Entry<String, Integer> word : freq.entrySet()) {
405 if (idfMap.containsKey(word.getKey())) {
406 int n = idfMap.get(word.getKey());
407 idfMap.put(word.getKey(), n + 1);
408 } else idfMap.put(word.getKey(), 1);
409 }
410 IDF.put(headersItems[j], idfMap);
411 }
412
413 // si el elemento es categórico múltiple
414 if (atr instanceof AtributoCategoricoMultiple) {
415 ((AtributoCategoricoMultiple) atr).setBitSetSize(bitsTable.size());
416 // ponemos a 1 los bits que representen los strings del atributo
417 for (String s : ((AtributoCategoricoMultiple) atr).getValor()) {
418 ((AtributoCategoricoMultiple) atr).setBitSetValue(bitsTable.get(s));
419 }
420 }
421
422 // si el elemento es númerico o de fecha
423 if (atr instanceof AtributoNumerico || atr instanceof AtributoFecha) {
424 // norm = (num - min) / (max - min)
425 double valorNormalizado;
426
427 if (atr instanceof AtributoNumerico) {
428 // el caso de las series en el que hay un valor erróneo
429 if (((AtributoNumerico) atr).getValor() == Double.MAX_VALUE) {
430 ((AtributoNumerico) atr).setValorNormalizado(-1.0);
431 } else {
432 double minNum = auxNormalizeNumeros[j][0];
433 double maxNum = auxNormalizeNumeros[j][1];
434 valorNormalizado = (((AtributoNumerico) atr).getValor() - minNum) / (maxNum - minNum);
435 ((AtributoNumerico) atr).setValorNormalizado(valorNormalizado);
436 }
437 }
438
439 if (atr instanceof AtributoFecha) {
440 long minLong = auxNormalizeFechas[j][0];
441 long maxLong = auxNormalizeFechas[j][1];
442 valorNormalizado = (double) (((AtributoFecha) atr).getValor() - minLong) / (maxLong - minLong);
443 ((AtributoFecha) atr).setValorNormalizado(valorNormalizado);
444 }
445
446 // actualizamos el ítem
447 it.setAtributo(headersItems[j], atr);
448 }
449 }
450
451 // actualizamos el array
452 arr.set(i, it);
453 }
454 items.setItems(arr);
455 }

465 private static Atributo transformarAtributo(int i) throws ParseException {
466 switch (types[i]) {
467 case "bool":
468 return new AtributoBoolean(words[i]);
469 case "string":
470 return new AtributoCategorico(words[i]);
471 case "double":
472 Atributo atrNum = new AtributoNumerico(words[i]);
473 if (!words[i].equals("") && !words[i].equals("-9223372036854775808")) {
474 auxNormalizeNumeros[i][0] = Math.min(auxNormalizeNumeros[i][0], (Double) atrNum.getValor());
475 auxNormalizeNumeros[i][1] = Math.max(auxNormalizeNumeros[i][1], (Double) atrNum.getValor());
476 } else atrNum = new AtributoNumerico(String.valueOf(Double.MAX_VALUE));
477 return atrNum;
478 case "set":
479 String[] list = words[i].split(";");
480 for (String s : list) {
481 if (!bitsTable.containsKey(s)) {
482 bitsTable.put(s, bitIndex);
483 bitIndex++;
484 }
485 }
486 return new AtributoCategoricoMultiple(words[i]);
487 case "fecha":
488 Atributo atrFecha = new AtributoFecha(words[i]);
489 auxNormalizeFechas[i][0] = Math.min(auxNormalizeFechas[i][0], (Long) atrFecha.getValor());
490 auxNormalizeFechas[i][1] = Math.max(auxNormalizeFechas[i][1], (Long) atrFecha.getValor());
491 return atrFecha;
492 case "texto":
493 return new AtributoTexto(words[i]);
494 default:
495 return null;
496 }

507 public static void processItems(ArrayList<String> data) throws Exception {
508 // obtenemos la cabecera y los tipos
509 headersItems = data.get(0).split(",");
510
511 auxNormalizeNumeros = new Double[headersItems.length][2];
512 auxNormalizeFechas = new Long[headersItems.length][2];
513
514 // inicializamos la estructura de datos al máximo y mínimo valor respectivamente
515 for (int i = 0; i < headersItems.length; i++) {
516 auxNormalizeNumeros[i][0] = Double.MAX_VALUE;
517 auxNormalizeNumeros[i][1] = Double.MIN_VALUE;
518 auxNormalizeFechas[i][0] = Long.MAX_VALUE;
519 auxNormalizeFechas[i][1] = Long.MIN_VALUE;
520 }
521
522 int itemId = 0;
523 for (int line = 1; line < data.size(); line++) {
524 words = data.get(line).split(validRegex(), -1);
525
526 // if para tratar los saltos de linea
527 if (words.length != headersItems.length) {
528 String actualLine = data.get(line) + data.get(line + 1);
529 words = actualLine.split(validRegex(), -1);
530 data.set(line + 1, actualLine);
531 } else {
532 // iteramos sobre todos elementos de la cabecera y lo asignamos con su tipo
533 // correspondiente
534 Item item = new Item(itemId, new HashMap<>());
535 for (int i = 0; i < headersItems.length; i++) {
536 if (!types[i].equals("other")) {
537 Atributo atr = transformarAtributo(i);
538 IDF.put(headersItems[i], new HashMap<>());
539
540 // guardamos el "id" original para posteriormente traducir
541 if (headersItems[i].equals("id") && !words[i].equals("")) {
542 itemsTable.put(Integer.parseInt(words[i]), itemId);
543 item.setOriginalId(Integer.parseInt(words[i]));
544 }
545 item.setAtributo(headersItems[i], atr);
546 }
547 }
548 itemId++;
549
550 // añadimos el ítem
551 items.addItem(item);
552 }
553 }
554 normalizarItems();
555 }

567 public static Usuarios processUsers(ArrayList<String> data, Path op) throws Exception {
568 Usuarios users = new Usuarios();
569
570 // generamos su id manualmente
571 int userId = 0;
572
573 String[] headersUsers = data.get(0).split(",");
574
575 for (int line = 1; line < data.size(); line++) {
576 words = data.get(line).split(validRegex(), -1);
577
578 int originalId = 0;
579 int originalItemId = 0;
580 double score = 0;
581
582 for (int i = 0; i < headersUsers.length; i++) {
583 // enlazamos cada elemento de la cabecera
584 if (headersUsers[i].contains("user")) originalId = Integer.parseInt(words[i]);
585 if (headersUsers[i].contains("item")) originalItemId = Integer.parseInt(words[i]);
586 if (headersUsers[i].equals("rating")) score = Double.parseDouble(words[i]);
587 }
588
589 // obtenemos el "id" del ítem generado manualmente mediante itemsTable
590 int itemId = itemsTable.get(originalItemId);
591 Item it = items.getItem(itemId);
592
593 boolean inList = true; // variable usada para comprobar que un usuario esté en el array
594 Map<Integer, Double> valoraciones = new HashMap<>();
595
596 if (usersTable.containsKey(originalId)) {
597 if (users.containsUser(usersTable.get(originalId)))
598 valoraciones = users.getUser(usersTable.get(originalId)).getValoraciones();
599 else inList = false;
600 } else {
601 usersTable.put(originalId, userId);
602 inList = false;
603 }
604
605 // añadimos la valoración
606 valoraciones.put(itemId, score);
607
608 if (maxScore < score) maxScore = score;
609 if (minScore > score) minScore = score;
610
611 // actualizamos el ítem
612 items.setItem(itemId, it);
613
614 // creamos un nuevo usuario con los ítems
615 Usuario user = new Usuario(usersTable.get(originalId), valoraciones);
616 user.setOriginalUserId(originalId);
617
618 // añadimos el usuario si no estaba, sino simplemente lo modificamos
619 if (inList) users.setUser(usersTable.get(originalId), user);
620 else {
621 users.addUser(user);
622 userId++;
623 }
624 }

626 return users;

637 public static void loadItems(ArrayList<String> data) throws Exception {
638 items = new Items();
639
640 // variables que iremos reutilizando
641 int itemId = 0, originalId = 0, size = 0;
642 Map<String, Atributo> atrs = new HashMap<>();
643
644 ArrayList<String> headers = new ArrayList<>();
645 ArrayList<String> tipos = new ArrayList<>();
646
647 ArrayList<ArrayList<Double>> tablaNormNums = new ArrayList<>();
648 ArrayList<ArrayList<Long>> tablaNormFechas = new ArrayList<>();
649
650 boolean finishedHeaders = false;
651 String typeOfData = "BITSTABLE";
652 for (String datum : data) {
653 String[] dataLine = datum.split("\\|");
654
655 // cambiamos el modo de lectura segun el titulo
656 if (datum.equals("ITEMS") || datum.equals("NORMITEMS") || datum.equals("NORMFECHAS")) {
657 typeOfData = datum;
658 continue;
659 }
660
661 // segun el tipo
662 if (datum.equals("#")) {
663 finishedHeaders = true;
664
665 // creamos el objeto
666 Item it = new Item(itemId, atrs);
667 it.setOriginalId(originalId);
668 items.addItem(it);
669
670 itemId++;
671 atrs = new HashMap<>();
672 } else if (typeOfData.equals("BITSTABLE")) {
673 String[] splitedData = datum.split(";");
674 if (splitedData.length > 1) bitsTable.put(splitedData[0], Integer.parseInt(splitedData[1]));
675 } else if (typeOfData.equals("NORMITEMS")) {
676 String[] splitedData = datum.split(";");
677 ArrayList<Double> vals = new ArrayList<>();
678 vals.add(Double.parseDouble(splitedData[0]));
679 vals.add(Double.parseDouble(splitedData[1]));
680 tablaNormNums.add(vals);
681 } else if (typeOfData.equals("NORMFECHAS")){
682 String[] splitedData = datum.split(";");
683 ArrayList<Long> vals = new ArrayList<>();
684 vals.add(Long.parseLong(splitedData[0]));
685 vals.add(Long.parseLong(splitedData[1]));
686 tablaNormFechas.add(vals);
687 } else if (typeOfData.equals("ITEMS")) {
688 if (size == 0) size = bitsTable.size();
689 if (!finishedHeaders) {
690 headers.add(dataLine[0]);
691 }
692 String tipoAtr = dataLine[1];
693
694 // tenemos en cuenta el id original
695 if (dataLine[0].equals("id")) {
696 originalId = Integer.parseInt(dataLine[2]);
697 itemsTable.put(originalId, itemId);
698 }
699
700 // dividimos y instanciamos los atributos segun el archivo
701 switch (tipoAtr) {
702 case "AtributoNumerico":
703 AtributoNumerico atrNum = new AtributoNumerico(dataLine[2]);
704 double valueNorm = Double.parseDouble(dataLine[3]);
705 atrNum.setValorNormalizado(valueNorm);
706 atrs.put(dataLine[0], atrNum);
707 if (!finishedHeaders) { tipos.add("double"); }
708 break;
709 case "AtributoBoolean":
710 AtributoBoolean atrBool = new AtributoBoolean(dataLine[2]);
711 atrs.put(dataLine[0], atrBool);
712 if (!finishedHeaders) { tipos.add("bool"); }
713 break;
714 case "AtributoFecha":
715 AtributoFecha atrFecha = new AtributoFecha(Long.parseLong(dataLine[2]));
716 valueNorm = Double.parseDouble(dataLine[3]);
717 atrFecha.setValorNormalizado(valueNorm);
718 atrs.put(dataLine[0], atrFecha);
719 if (!finishedHeaders) { tipos.add("fecha"); }
720 break;
721 case "AtributoCategorico":
722 AtributoCategorico atrCat = new AtributoCategorico("");
723 if (dataLine.length > 2)
724 atrCat = new AtributoCategorico(dataLine[2]);
725 atrs.put(dataLine[0], atrCat);
726 if (!finishedHeaders) { tipos.add("string"); }
727 break;
728 case "AtributoCategoricoMultiple":
729 AtributoCategoricoMultiple atrCatMult = new AtributoCategoricoMultiple("");
730 // primero convertimos a set
731 if (dataLine.length > 2) {
732 String dataLineFixed = dataLine[2].replace("[", "").replace("]", "").replace(", ", ";");
733 atrCatMult = new AtributoCategoricoMultiple(dataLineFixed);
734
735 //ahora convertimos el bitset
736 if (dataLine.length > 3) {
737 dataLineFixed = dataLine[3].replace("{", "").replace("}", "").replace(", ", ";");
738 String[] valores = dataLineFixed.split(";");
739 atrCatMult.setBitSetSize(size);
740 for (String val : valores) {
741 int n = Integer.parseInt(val);
742 atrCatMult.setBitSetValue(n);
743 }
744 int idx = 0;
745 }
746 }
747 atrs.put(dataLine[0], atrCatMult);
748 if (!finishedHeaders) { tipos.add("set"); }
749 break;
750 case "AtributoTexto":
751 String dataLineFixed = (dataLine[2].replace("{", "")).replace("}", "");
752 Map<String, Integer> valoresMap = new HashMap<>();
753 String[] valores = dataLineFixed.split(",");
754
755 // convertimos el map
756 for (String line : valores) {
757 String[] vals = line.split("=");
758 if (vals.length > 1) valoresMap.put(vals[0].trim(), Integer.parseInt(vals[1]));
759 }
760 AtributoTexto atrText = new AtributoTexto(valoresMap, Integer.parseInt(dataLine[3]));
761 atrs.put(dataLine[0], atrText);
762 if (!finishedHeaders) { tipos.add("texto"); }
763 break;
764 }
765 }
766 }
767
768 // convertimso arrayList a arrays
769 headersItems = headers.toArray(new String[0]);
770 types = tipos.toArray(new String[0]);
771
772 // convertimos los arrayList a las matrices
773 auxNormalizeNumeros = new Double[tablaNormNums.size()][2];
774 auxNormalizeFechas = new Long[tablaNormFechas.size()][2];
775 for (int i = 0; i < tablaNormNums.size(); i++) {
776 auxNormalizeNumeros[i][0] = tablaNormNums.get(i).get(0);
777 auxNormalizeNumeros[i][1] = tablaNormNums.get(i).get(1);
778 auxNormalizeFechas[i][0] = tablaNormFechas.get(i).get(0);
779 auxNormalizeFechas[i][1] = tablaNormFechas.get(i).get(1);
780 }
