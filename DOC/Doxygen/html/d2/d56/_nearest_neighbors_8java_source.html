92 public double cosineSimilarityBitSet(BitSet bsA, BitSet bsB) {
93 BitSet combination = (BitSet) bsA.clone();
94 combination.and(bsB);
95
96 double dotProduct = combination.cardinality();
97 double normA = bsA.cardinality();
98 double normB = bsB.cardinality();
99
100 if (normA == 0 || normB == 0) return 1;
101 return 1 - dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
102 }

113 public double cosineSimilarity(Map<String, Double> mapA, Map<String, Double> mapB) {
114 double numerador = 0;
115 double denominadorA = 0;
116 double denominadorB = 0;
117 for (Map.Entry<String, Double> words : mapA.entrySet()) {
118 if (mapB.containsKey(words.getKey())) {
119 numerador += words.getValue() * mapB.get(words.getKey()).doubleValue();
120 }
121 denominadorA += Math.pow(words.getValue(), 2);
122 }
123 for (Map.Entry<String, Double> words : mapB.entrySet()) {
124 denominadorB += Math.pow(words.getValue(), 2);
125 }
126 if (denominadorA == 0 || denominadorB == 0) return 1;
127 return 1 - numerador / (Math.sqrt(denominadorA)*Math.sqrt(denominadorB));
128 }

261 public Double prediccionValoracionItem(Item item) {
262 double predValoracion = 0;
263 Map <Integer, Double> valoraciones = super.u.getValoraciones();
264 for (Map.Entry<Integer, Double> valoracionUsuario : valoraciones.entrySet()) {
265 Item knownItem = Parser.getAllItems().getItem(valoracionUsuario.getKey());
266 Map<String, Atributo> attributes = knownItem.getAtributos();
267 double distance = 0;
268 for (Map.Entry<String, Atributo> att : attributes.entrySet()) {
269 if (att.getValue() instanceof AtributoBoolean) distance += distanceBooleans(att.getKey(), knownItem, item);
270 else if (att.getValue() instanceof AtributoNumerico) distance += distanceNums(att.getKey(), knownItem, item);
271 else if (att.getValue() instanceof AtributoCategorico) distance += distanceCategorico(att.getKey(), knownItem, item);
272 else if (att.getValue() instanceof AtributoCategoricoMultiple) distance += distanceCategoricoMultiple(att.getKey(), knownItem, item);
273 else if (att.getValue() instanceof AtributoTexto) distance += distanceTexto(att.getKey(), knownItem, item);
274 else if (att.getValue() instanceof AtributoFecha) distance += distanceFechas(att.getKey(), knownItem, item);
275 }
276 distance /= attributes.size();
277 predValoracion += ((1-distance) * valoracionUsuario.getValue());
278 }
279 return predValoracion;
